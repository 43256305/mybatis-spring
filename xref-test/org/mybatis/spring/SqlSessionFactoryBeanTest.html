259   @Test
260   void testSetConfigLocation() throws Exception {
261     setupFactoryBean();
262 
263     factoryBean
264         .setConfigLocation(new org.springframework.core.io.ClassPathResource("org/mybatis/spring/mybatis-config.xml"));
265 
266     SqlSessionFactory factory = factoryBean.getObject();
267 
268     assertThat(factory.getConfiguration().getEnvironment().getId())
269         .isEqualTo(SqlSessionFactoryBean.class.getSimpleName());
270     assertThat(factory.getConfiguration().getEnvironment().getDataSource()).isSameAs(dataSource);
271     assertThat(factory.getConfiguration().getEnvironment().getTransactionFactory().getClass())
272         .isSameAs(org.mybatis.spring.transaction.SpringManagedTransactionFactory.class);
273     assertThat(factory.getConfiguration().getVfsImpl()).isSameAs(JBoss6VFS.class);
274 
275     // properties explicitly set differently than the defaults in the config xml
276     assertThat(factory.getConfiguration().isCacheEnabled()).isFalse();
277     assertThat(factory.getConfiguration().isUseGeneratedKeys()).isTrue();
278     assertThat(factory.getConfiguration().getDefaultExecutorType())
279         .isSameAs(org.apache.ibatis.session.ExecutorType.REUSE);
280 
281     // for each statement in the xml file: org.mybatis.spring.TestMapper.xxx & xxx
282     assertThat(factory.getConfiguration().getMappedStatementNames().size()).isEqualTo(8);
283 
284     assertThat(factory.getConfiguration().getResultMapNames().size()).isEqualTo(0);
285     assertThat(factory.getConfiguration().getParameterMapNames().size()).isEqualTo(0);
286   }

495   private void assertDefaultConfig(SqlSessionFactory factory) {
496     assertConfig(factory, SqlSessionFactoryBean.class.getSimpleName(),
497         org.mybatis.spring.transaction.SpringManagedTransactionFactory.class);
498     assertThat(factory.getConfiguration().getVariables().size()).isEqualTo(0);

501   private void assertConfig(SqlSessionFactory factory, Class<? extends TransactionFactory> transactionFactoryClass) {
502     assertConfig(factory, SqlSessionFactoryBean.class.getSimpleName(), transactionFactoryClass);
503   }
504 
505   private void assertConfig(SqlSessionFactory factory, String environment,
506       Class<? extends TransactionFactory> transactionFactoryClass) {
507     assertThat(factory.getConfiguration().getEnvironment().getId()).isEqualTo(environment);
508     assertThat(factory.getConfiguration().getEnvironment().getDataSource()).isSameAs(dataSource);
509     assertThat(factory.getConfiguration().getEnvironment().getTransactionFactory().getClass())
510         .isSameAs(transactionFactoryClass);
511 
512     // no mappers configured => no mapped statements or other parsed elements
513     assertThat(factory.getConfiguration().getMappedStatementNames().size()).isEqualTo(0);
514     assertThat(factory.getConfiguration().getResultMapNames().size()).isEqualTo(0);
515     assertThat(factory.getConfiguration().getParameterMapNames().size()).isEqualTo(0);
516     assertThat(factory.getConfiguration().getSqlFragments().size()).isEqualTo(0);
517   }
